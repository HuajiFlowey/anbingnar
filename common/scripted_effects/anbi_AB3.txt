anbi_is_merfolk_state_effect = {
	if = {
		limit = { 
			owner = {
				has_government_attribute = anbi_is_merfolk_state
			}
			OR = {
				AND = {
					NOT = {
						owner = {
							mission_completed = anbi_merfolk_djinnakah
						}
					}
					superregion = djinnakah_superregion
				}
				AND = {
					NOT = {
						owner = {
							mission_completed = anbi_merfolk_the_empire_of_land_and_sea
						}
					}
					OR = {
						superregion = taneyas_superregion
						region = fahvanosy_region
					}					
				}
				NOT = {
					superregion = djinnakah_superregion
					superregion = taneyas_superregion
					region = fahvanosy_region
				}
			}			
		}
		add_province_modifier = {
